#GCW1 — Shiny App: Tree-Based Methods for Heart Disease
#Dataset: Cleveland Heart Disease Dataset (UCI / Kaggle)

#This application predicts the presence of heart disease using
#two tree-based machine learning models:
#1. Decision Tree
#2. Random Forest

#The Shiny interface allows users to:
#• Select the model
#• Adjust model parameters
#• Visualise model structure
#• Evaluate predictive performance

#The aim is to demonstrate how interpretable ML methods can be
#applied to healthcare data.

# LOAD REQUIRED LIBRARIES
# shiny        → Build interactive web applications in R
# rpart        → Decision tree algorithm
# rpart.plot   → Visualisation of decision trees
# randomForest → Ensemble tree-based model
# ggplot2      → General plotting library
# dplyr        → Data manipulation
# corrplot     → Correlation visualisation (optional)

library(shiny)
library(rpart)
library(rpart.plot)
library(randomForest)
library(ggplot2)
library(dplyr)
library(corrplot)

#LOAD AND CLEAN DATA

#Read the Cleveland heart disease dataset.
#Missing values are coded as "?" and converted to NA.

heart <- read.csv(
  "processed.cleveland.data.csv",
  header = FALSE,
  na.strings = "?"
)

#Assign meaningful column names based on dataset documentation

colnames(heart) <- c(
  "age",       # Age in years
  "sex",       # Biological sex
  "cp",        # Chest pain type
  "trestbps",  # Resting blood pressure
  "chol",      # Serum cholesterol
  "fbs",       # Fasting blood sugar > 120 mg/dl
  "restecg",   # Resting ECG results
  "thalach",   # Maximum heart rate achieved
  "exang",     # Exercise-induced angina
  "oldpeak",   # ST depression induced by exercise
  "slope",     # Slope of peak exercise ST segment
  "ca",        # Number of major vessels
  "thal",      # Thalassemia test result
  "target"     # Heart disease diagnosis
)

#Convert target to binary classification:
#0 = no disease, 1 = presence of disease

heart$target <- ifelse(heart$target > 0, 1, 0)

#Remove rows containing missing values to ensure models
#are trained on complete observations

heart <- na.omit(heart)



#CONVERT VARIABLE TYPES

#Categorical variables must be factors so that tree models
#treat them as discrete predictors rather than numeric values

factor_vars <- c(
  "sex", "cp", "fbs", "restecg",
  "exang", "slope", "ca", "thal", "target"
)

heart[factor_vars] <- lapply(heart[factor_vars], as.factor)

#Ensure continuous variables are numeric (safety step)

numeric_vars <- c("age", "trestbps", "chol", "thalach", "oldpeak")

for (col in numeric_vars) {
  heart[[col]] <- suppressWarnings(
    as.numeric(as.character(heart[[col]]))
  )
}

#Remove any rows that became NA during conversion

heart <- na.omit(heart)


#TRAIN / TEST SPLIT

#Split the dataset into:
#• Training set (70%) for model fitting
#• Test set (30%) for unbiased evaluation

set.seed(123)  # Ensures reproducibility

train_index <- sample(
  seq_len(nrow(heart)),
  size = 0.7 * nrow(heart)
)

train <- heart[train_index, ]
test  <- heart[-train_index, ]


#SHINY USER INTERFACE

ui <- fluidPage(
  
  titlePanel("Heart Disease Prediction Using Tree-Based Methods"),
  
  sidebarLayout(
    
    sidebarPanel(
      
      h3("Model Controls"),
      
      #User selects which model to use
      selectInput(
        "model_type",
        "Choose Model:",
        choices = c("Decision Tree", "Random Forest")
      ),
      
      #Controls pruning strength for the decision tree
      sliderInput(
        "cp",
        "Tree Complexity (Decision Tree)",
        min = 0.001,
        max = 0.1,
        value = 0.01,
        step = 0.005
      ),
      
      #Controls number of trees in the random forest
      numericInput(
        "ntree",
        "Number of Trees (Random Forest)",
        value = 100,
        min = 50,
        max = 500
      )
    ),
    
    mainPanel(
      
      tabsetPanel(
        
        #Visual output of models
        tabPanel(
          "Model Visualisation",
          plotOutput("modelPlot")
        ),
        
        #Performance evaluation
        tabPanel(
          "Model Performance",
          verbatimTextOutput("performance")
        ),
        
        #Explanatory text for non-technical users
        tabPanel(
          "Explanation",
          
          h4("How Tree-Based Methods Work"),
          
          p("Decision trees classify patients using a sequence of rules."),
          p("Random forests combine many trees to improve accuracy and reduce overfitting."),
          
          h4("Key Predictors of Heart Disease"),
          verbatimTextOutput("insights")
        )
      )
    )
  )
)



#SHINY SERVER LOGIC

server <- function(input, output) {
  

#DECISION TREE MODEL

  
  tree_model <- reactive({
    
    #Train a decision tree classifier
    #cp controls pruning to avoid overfitting
    
    rpart(
      target ~ .,
      data = train,
      method = "class",
      control = rpart.control(cp = input$cp)
    )
  })
  
  
 
#RANDOM FOREST MODEL
  
  
  rf_model <- reactive({
    
    #Train an ensemble of decision trees.
    #Importance = TRUE enables feature importance scores.
    
    randomForest(
      target ~ .,
      data = train,
      ntree = input$ntree,
      importance = TRUE
    )
  })
  
  
#MODEL VISUALISATION

  
  output$modelPlot <- renderPlot({
    
    if (input$model_type == "Decision Tree") {
      
      #Display tree structure
      
      rpart.plot(
        tree_model(),
        type = 2,
        extra = 104,
        fallen.leaves = TRUE,
        main = "Decision Tree for Heart Disease"
      )
      
    } else {
      
      #Display variable importance for Random Forest
      
      varImpPlot(
        rf_model(),
        main = "Variable Importance (Random Forest)"
      )
    }
  })
  
  

#MODEL PERFORMANCE
  
  output$performance <- renderPrint({
    
    #Generate predictions on unseen test data
    
    if (input$model_type == "Decision Tree") {
      preds <- predict(tree_model(), test, type = "class")
    } else {
      preds <- predict(rf_model(), test)
    }
    
    #Confusion matrix summarises classification results
    
    cm <- table(Predicted = preds, Actual = test$target)
    
    cat("Confusion Matrix:\n\n")
    print(cm)
    
    #Calculate overall accuracy
    
    accuracy <- sum(diag(cm)) / sum(cm)
    
    cat("\nOverall Accuracy:", round(accuracy, 3))
  })
  
  

#VARIABLE IMPORTANCE
  
  output$insights <- renderPrint({
    
    if (input$model_type == "Random Forest") {
      
      #Extract importance scores
      
      imp <- importance(rf_model())
      
      imp <- imp[complete.cases(imp), , drop = FALSE]
      
      # Identify top predictors
      
      top_vars <- names(
        sort(imp[, "MeanDecreaseGini"], decreasing = TRUE)
      )[1:3]
      
      cat("The three most important predictors are:\n\n")
      print(top_vars)
      
      cat("\nThese variables have the strongest influence on predictions.")
      
    } else {
      
      cat("Variable importance is available only for Random Forest.\n")
      cat("For Decision Trees, you can interpret important features by 
          examining the tree structure in the 'Model Visualisation' tab. 
          Features near the top of the tree generally have the largest 
          influence on the prediction")
    }
  })
}


#RUN SHINY APPLICATION
shinyApp(ui = ui, server = server)
